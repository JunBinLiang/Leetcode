class Solution {
    public int smallestDistancePair(int[] nums, int k) {
        List<Integer> list=new LinkedList();
        for(int i=0;i<nums.length;i++){
            for(int j=i+1;j<nums.length;j++){
                list.add(Math.abs(nums[j]-nums[i]));
            }
        }
        Collections.sort(list);
        return list.get(k-1);
    }
}
